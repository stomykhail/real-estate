import psycopg2
import time

# Dockerized Database connection parameters
DB_HOST = "localhost"
DB_PORT = "5432"
DB_NAME = "postgres"
DB_USER = "postgres"
DB_PASSWORD = "mysecretpassword"

# Path to the clean CSV you generated earlier
CSV_FILE_PATH = "/tmp/real_estate_csvs/Realestate_ready_for_db.csv"

def load_fact_table():
    print(f"Connecting to PostgreSQL at {DB_HOST}...")
    
    try:
        # 1. Establish the connection to the Docker database
        conn = psycopg2.connect(
            host=DB_HOST,
            port=DB_PORT,
            dbname=DB_NAME,
            user=DB_USER,
            password=DB_PASSWORD
        )
        conn.autocommit = True
        cursor = conn.cursor()

        print(f"Preparing to bulk load 1.1M rows from {CSV_FILE_PATH}...")
        start_time = time.time()

        # 2. Use copy_expert to trigger Postgres's blazing fast bulk-loader
        with open(CSV_FILE_PATH, 'r') as f:
            # The SQL COPY command. Notice we skip the header row.
            copy_sql = """
                COPY fact_real_estate_transaction (
                    transaction_id, serial_number, assessed_value, 
                    sales_amount, sales_ratio, address_id, 
                    property_type_id, date_recorded_id
                ) FROM STDIN WITH CSV HEADER DELIMITER ',';
            """
            cursor.copy_expert(sql=copy_sql, file=f)

        end_time = time.time()
        load_time = round(end_time - start_time, 2)
        
        # 3. Verify the load
        cursor.execute("SELECT COUNT(*) FROM fact_real_estate_transaction;")
        row_count = cursor.fetchone()[0]
        
        print(f"SUCCESS: Loaded {row_count} rows in {load_time} seconds!")

    except Exception as e:
        print(f"ERROR: Failed to load data. Details: {e}")
    
    finally:
        # Always close your connections professionally!
        if 'conn' in locals():
            cursor.close()
            conn.close()
            print("Database connection closed.")

if __name__ == "__main__":
    load_fact_table()
